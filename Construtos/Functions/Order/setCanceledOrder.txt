CREATE OR REPLACE PROCEDURE setCanceledOrder(
  IN order_id INTEGER )
AS $$
BEGIN
       if order_id is not null then
            update appxptostore_order
            set "isCanceled"=TRUE
            where id=order_id;
        else
               raise exception 'NÃ£o pode ter valores nulos!';
        end if;
END
$$ language plpgsql;
